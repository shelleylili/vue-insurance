<template>
  <div class="step1">
    <!-- 身份证回显card -->
    <group>
      <cell title="姓名" :value="idInfos.name"></cell>
      <cell title="身份证号" :value="idInfos.code"></cell>
    </group>
    <!--下一步 -->
    <Button :clickHandler="next" :text="text"></Button>
  </div>
</template>

<script>
  import { Cell, Group } from 'vux'
  import Button from '../../chunks/Button'
  import bus from '../../../bus'
 
  export default {
    name: 'Step1',
    data () {
      return {
        text: '下一步', // 下一步按钮文字
        idInfos: {}
      }
    },
    created () {
      bus.$on('realChecked', (user) => {
        this.idInfos = user.idInfos
        let starCount = this.idInfos.code.length - 10
        let str = ''
        for (let i = 0; i < starCount; i++) {
          str += '*'
        }
        this.idInfos.code = this.idInfos.code.substr(0, 6) + str + this.idInfos.code.substr(-4, 4)
      })
    },
    methods: {
      next: function () {
        this.$router.push('/step2')
        bus.$emit('fastpy2', '理赔凭证')
      }
    },
    components: {
      Group,
      Button,
      Cell
    }
  }
</script>
